typedef struct {
    float x, y, z;
} position_t;

typedef struct {
    float health;
} health_t;

void my_system(nuecs_world_t world, nuecs_chunk_t chunk) 
{
    health_t    *healths   = nuecs_chunk_get(chunk, 0);
    positions_t *positions = nuecs_chunk_get(chunk, 1);
    uint32_t count = nuecs_chunk_get_size(chunk);
    for (uint32_t i = 0; i < count; i++) {
        healths[i] += positions[i];
    }
}
